df_prep = df.copy()

df_prep['Va de acceso'] = np.where(df_prep['Va de acceso'] == 'Aeropuerto', 'Aeropuerto', 'Otros')
df_prep['Motivo del viaje'] = 'Ocio'  # Prcticamente constante, se descarta despus

freq_ccaa = df_prep['Comunidad autnoma de destino'].value_counts(normalize=True)
ccaa_bajas = freq_ccaa[freq_ccaa < 0.01].index
df_prep['Comunidad autnoma de destino'] = df_prep['Comunidad autnoma de destino'].replace(ccaa_bajas, 'Otras')

freq_pais = df_prep['Pas de residencia'].value_counts(normalize=True)
paises_bajos = freq_pais[freq_pais < 0.01].index
df_prep['Pas de residencia'] = df_prep['Pas de residencia'].replace(paises_bajos, 'Otros')

orden_duracion = {
    '1 noche': 1,
    'De 2 a 3 noches': 2,
    'De 4 a 7 noches': 3,
    'De 8 a 15 noches': 4,
    'Ms de 15 noches': 5
}
df_prep['Duracin codificada'] = df_prep['Duracin del viaje'].map(orden_duracion)

df_prep['Total_log'] = np.log1p(df_prep['Total'])

def cramers_v(x, y):
    confusion_matrix = pd.crosstab(x, y)
    chi2 = ss.chi2_contingency(confusion_matrix)[0]
    n = confusion_matrix.sum().sum()
    phi2 = chi2 / n
    r, k = confusion_matrix.shape
    phi2corr = max(0, phi2 - ((k - 1)*(r - 1)) / (n - 1))
    rcorr = r - ((r - 1)**2) / (n - 1)
    kcorr = k - ((k - 1)**2) / (n - 1)
    return np.sqrt(phi2corr / min((kcorr - 1), (rcorr - 1)))

cat_vars = [
    'Va de acceso', 'Comunidad autnoma de destino', 'Tipo de alojamiento',
    'Forma de organizacin del viaje', 'Pas de residencia'
]

matriz_cramer = pd.DataFrame(index=cat_vars, columns=cat_vars)

for v1, v2 in itertools.combinations(cat_vars, 2):
    val = cramers_v(df_prep[v1], df_prep[v2])
    matriz_cramer.loc[v1, v2] = round(val, 3)
    matriz_cramer.loc[v2, v1] = round(val, 3)

print("Matriz de asociacin Cramers V (01):")
display(matriz_cramer)

In[5]:
vars_finales = [
    'Comunidad autnoma de destino', 'Pas de residencia',
    'Va de acceso', 'Tipo de alojamiento',
    'Forma de organizacin del viaje', 'Duracin codificada', 'Total_log'
]
df_final = df_prep[vars_finales].copy()

cat_vars = [
    'Comunidad autnoma de destino', 'Pas de residencia',
    'Va de acceso', 'Tipo de alojamiento', 'Forma de organizacin del viaje'
]

encoder = OneHotEncoder(sparse_output=False)
encoded = encoder.fit_transform(df_final[cat_vars])
encoded_cols = encoder.get_feature_names_out(cat_vars)

df_encoded = pd.DataFrame(encoded, columns=encoded_cols, index=df_final.index)

df_cluster_ready = pd.concat([
    df_encoded,
    df_final[['Duracin codificada', 'Total_log']]
], axis=1)

print("Dataset numrico listo para clustering:", df_cluster_ready.shape)

scaler = StandardScaler()
df_scaled = scaler.fit_transform(df_cluster_ready)

pca = PCA(n_components=2, random_state=42)
pca_result = pca.fit_transform(df_scaled)

pca_df = pd.DataFrame(pca_result, columns=['PC1', 'PC2'])
pca_df['Comunidad'] = df_prep['Comunidad autnoma de destino']
pca_df['Pas'] = df_prep['Pas de residencia']

plt.figure(figsize=(10, 6))
sns.scatterplot(data=pca_df, x='PC1', y='PC2', hue='Comunidad', alpha=0.7, s=50, legend=False)
plt.title("PCA Distribucin de registros por Comunidad Autnoma")
plt.show()

df_cluster_ready.to_csv("df_clustering.csv", index=False, encoding='utf-8-sig')
print("Archivo guardado como df_clustering.csv")
